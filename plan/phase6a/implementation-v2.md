## 1. Settings additions (`src/assistant/config.py`)

Add `uploads_dir` to the `Settings` class. The default differs container vs Mac dev: in container `/app/.uploads/` is the canonical path (RQ1-locked); on Mac dev `<data_dir>/uploads/` is the fallback (Mac dev cannot run the live PDF Read probe — develop extract path locally, smoke PDFs only via VPS).

Detection heuristic: if `project_root == Path("/app")`, we're in container.

```python
def _default_uploads_dir(project_root: Path, data_dir: Path) -> Path:
    """Resolve uploads tmp dir.

    Container: /app/.uploads/ (RQ1 Option 1 - keeps make_file_hook's
    single project_root constraint working untouched).
    Mac dev: <data_dir>/uploads/ - PDF native-Read probe unreachable
    off-VPS, but extract path develops locally.
    """
    if project_root == Path("/app"):
        return Path("/app/.uploads")
    return (data_dir / "uploads").expanduser().resolve()
```

In `Settings`, expose as a `@property` (mirrors existing `vault_dir` / `memory_index_path` patterns):


```python
async def _on_document(self, message: Message) -> None:
    """Phase 6a: file-attachment ingestion.

    Pre-download size guard handles aiogram's Optional[int] file_size
    safely (devil C2). Suffix whitelist + bot.download with
    TelegramBadRequest envelope for the >20MB-but-no-file_size case.

    Tmp path: <uploads_dir>/<uuid>__<sanitized_orig>.<ext>. Sanitization
    strips non-word chars and caps original at 64 chars (devil M5 - eases
    quarantine post-mortem).
    """
    if self._handler is None:
        log.warning("document_received_without_handler")
        return
    doc = message.document
    if doc is None:
        # F.document filter guarantees non-None; defensive for type-checker.
        return

    # Devil C2: file_size is Optional[int]; (x or 0) > N is type-safe.
    file_size = doc.file_size or 0
    if file_size > 20 * 1024 * 1024:
        await message.reply(
            "файл больше 20 МБ — это лимит Telegram bot API; пришли поменьше"
        )
        return

    # Devil M1: file_name is Optional[str]; need a usable suffix.
    file_name = doc.file_name or ""
    suffix = Path(file_name).suffix.lower().lstrip(".")
    if not suffix or suffix not in _SUFFIX_WHITELIST:
        await message.reply(
            "формат не поддерживается; список: PDF, DOCX, TXT, MD, XLSX"
        )
        return

    # Sanitize original name for tmp path: strip path-traversal chars,
    # control chars, punctuation; cap at 64 to keep filenames bounded.
    orig_stem = Path(file_name).stem or "attachment"
    safe_stem = re.sub(r"[^\w.-]", "_", orig_stem)[:64]
    tmp_name = f"{uuid4().hex}__{safe_stem}.{suffix}"
    uploads_dir = self._settings.uploads_dir
    uploads_dir.mkdir(parents=True, exist_ok=True)
    tmp_path = uploads_dir / tmp_name

    try:
        await self._bot.download(doc, destination=tmp_path)
    except TelegramBadRequest as exc:
        # Devil C2: TelegramBadRequest, NOT TelegramEntityTooLarge
        # (which is SEND-side). Telegram emits "file is too big" when
        # file_size is None but the actual file exceeds the API limit.
        msg_lower = str(exc).lower()
        if "too big" in msg_lower:
            await message.reply(
                "файл больше Telegram bot API лимита; пришли поменьше"
            )
        else:
            await message.reply(f"не смог скачать файл: {exc}")
        return

    # Construct IncomingMessage with attachment fields. Use caption (or
    # default Russian prompt for empty caption - devil L6 inclusive of
    # TXT/MD).
    caption = (message.caption or "").strip() or "опиши содержимое файла"
    incoming = IncomingMessage(
        chat_id=message.chat.id,
        message_id=message.message_id,
        text=caption,
        attachment=tmp_path,
        attachment_kind=suffix,  # type: ignore[arg-type]  # narrowed by whitelist.
        attachment_filename=file_name,
    )

    chunks: list[str] = []

    async def emit(text: str) -> None:
        chunks.append(text)

    async with ChatActionSender.typing(bot=self._bot, chat_id=message.chat.id):
        await self._handler.handle(incoming, emit)

    full = "".join(chunks).strip() or "(пустой ответ)"
    for part in _split_for_telegram(full, limit=TELEGRAM_MSG_LIMIT):
        await self._bot.send_message(message.chat.id, part)
```

**Note on the `# type: ignore[arg-type]`.** `attachment_kind` is `Literal["pdf", "docx", "txt", "md", "xlsx"] | None`; we just narrowed `suffix` to one of those values via the whitelist check - but mypy can't infer Literal narrowing from a `frozenset[str]` membership test. The type-ignore is local and self-documenting. (Coder may convert to `cast(Literal[...], suffix)` if preferred - same effect.)

### 4.4 New `_handle_extraction_failure`

Quarantines the file then emits a Russian reply.

```python
async def _handle_extraction_failure(
    self,
    msg: IncomingMessage,
    turn_id: int,
    exc: ExtractionError,
    emit: Emit,
) -> None:
    """Quarantine the failed file; emit Russian reply; mark turn complete.

    Quarantine path: <uploads_dir>/.failed/<orig_filename>. Boot-sweep
    prunes >7d old.
    """
    assert msg.attachment is not None
    quarantine_dir = self._settings.uploads_dir / ".failed"
    quarantine_dir.mkdir(parents=True, exist_ok=True)
    target = quarantine_dir / msg.attachment.name
    try:
        msg.attachment.rename(target)
    except OSError as rename_exc:
        log.warning(
            "quarantine_rename_failed",
            path=str(msg.attachment),
            error=repr(rename_exc),
        )
    log.info(
        "extraction_failed_quarantined",
        turn_id=turn_id,
        path=str(target),
        reason=str(exc),
    )
    reply = f"не смог прочитать файл: {exc}"
    if "encrypted" in str(exc).lower():
        reply = "файл зашифрован — пришли расшифрованный"
    await emit(reply)
    # Mark turn complete with a synthetic meta - no SDK call was made.
    await self._conv.complete_turn(
        turn_id,
        meta={"stop_reason": "extraction_error", "usage": {}},
    )
```

**Note:** the cleanup `finally` in `_handle_locked` checks `msg.attachment.exists()` before `unlink` - quarantine `rename` already moved the file, so the `unlink` is a no-op. Idempotent + safe.

## 5. `files/extract.py` module (NEW)

Path: `src/assistant/files/extract.py`. Empty `__init__.py` co-located.

```python
"""Phase 6a: file-attachment extractors.

Each extractor returns ``(extracted_text, char_count)`` or raises
:class:`ExtractionError`.

Caps:
- DOCX: no per-doc cap (the post-extract 200K cap in the handler is
  sufficient; document-order traversal is intrinsically bounded).
- XLSX: ``ROW_CAP=20``, ``COL_CAP=30`` per sheet (RQ2 Q13 retune); plus
  total ``POST_EXTRACT_CHAR_CAP=200_000`` on a clean sheet boundary.
- PDF: fallback only - used when the SDK Read tool's multimodal path
  fails or in pypdf-uniform mode.
"""

from __future__ import annotations

import logging
from pathlib import Path
from typing import Callable

log = logging.getLogger("files.extract")

POST_EXTRACT_CHAR_CAP = 200_000
XLSX_ROW_CAP = 20
XLSX_COL_CAP = 30
PDF_MIN_TEXT_LAYER_CHARS = 100


class ExtractionError(Exception):
    """Raised by an extractor when the file is unreadable.

    Reason categories used by the handler for user-facing messages:
    - "encrypted": password-protected file
    - "corrupt": truncated/malformed
    - "format": MIME-suffix mismatch caught at parse time
    """


def extract_docx(path: Path) -> tuple[str, int]:
    """Extract DOCX in document order (devil RQ4 fix).

    Walks <w:p> and <w:tbl> children of the body in source order,
    instead of paragraphs-then-tables (which mangles the model's
    reading sequence on interleaved layouts).
    """
    try:
        from docx import Document
        from docx.opc.exceptions import PackageNotFoundError
        from docx.oxml.ns import qn
    except ImportError as exc:
        raise ExtractionError(f"python-docx not installed: {exc}") from exc

    try:
        doc = Document(str(path))
    except PackageNotFoundError as exc:
        raise ExtractionError("encrypted or corrupted DOCX") from exc
    except (KeyError, ValueError) as exc:
        # Devil M7: mangled XML namespaces / custom XML.
        raise ExtractionError(f"corrupt DOCX: {exc}") from exc
    except Exception as exc:  # belt-and-suspenders for unknown parse errors
        raise ExtractionError(f"DOCX parse failed: {exc}") from exc

    out: list[str] = []
    try:
        for child in doc.element.body.iterchildren():
            if child.tag == qn("w:p"):
                text = "".join(t.text or "" for t in child.iter(qn("w:t")))
                if text.strip():
                    out.append(text)
            elif child.tag == qn("w:tbl"):
                for row in child.iter(qn("w:tr")):
                    cells = [
                        "".join(t.text or "" for t in cell.iter(qn("w:t")))
                        for cell in row.iter(qn("w:tc"))
                    ]
                    out.append("\t".join(cells))
    except Exception as exc:
        raise ExtractionError(f"DOCX traversal failed: {exc}") from exc

    text = "\n".join(out)
    return text, len(text)


def extract_xlsx(path: Path) -> tuple[str, int]:
    """Extract XLSX with ROW/COL caps and total-cap clean-boundary.

    All sheets, ROW_CAP=20 rows x COL_CAP=30 cols per sheet. Total cap
    POST_EXTRACT_CHAR_CAP enforced on sheet-boundary; defensive
    final-substring assert lives in the handler.

    read_only=True keeps RSS flat (RQ2: 42 MB on 308 MB worst-case).
    data_only=True surfaces cached formula values (devil L3 caveat:
    script-generated XLSX without an Excel save pass shows None for
    formula cells).
    wb.close() in finally is mandatory (openpyxl leaks fd otherwise).
    """
    try:
        from openpyxl import load_workbook
        from zipfile import BadZipFile
    except ImportError as exc:
        raise ExtractionError(f"openpyxl not installed: {exc}") from exc

    try:
        wb = load_workbook(filename=str(path), read_only=True, data_only=True)
    except BadZipFile as exc:
        raise ExtractionError("encrypted or corrupted XLSX") from exc
    except KeyError as exc:
        # openpyxl raises KeyError for missing required parts (corrupt zip).
        raise ExtractionError(f"corrupt XLSX: {exc}") from exc
    except Exception as exc:
        raise ExtractionError(f"XLSX parse failed: {exc}") from exc

    out: list[str] = []
    total_chars = 0
    try:
        for sheet_name in wb.sheetnames:
            chunk_lines = [f"=== Sheet: {sheet_name} ==="]
            ws = wb[sheet_name]
            row_count = 0
            for row in ws.iter_rows(
                values_only=True, max_row=XLSX_ROW_CAP, max_col=XLSX_COL_CAP
            ):
                row_count += 1
                cells = [
                    "" if v is None else str(v) for v in row[:XLSX_COL_CAP]
                ]
                chunk_lines.append("\t".join(cells))
                if row_count >= XLSX_ROW_CAP:
                    break
            chunk = "\n".join(chunk_lines)
            # Devil H1: clean sheet boundary on total-cap.
            if total_chars + len(chunk) > POST_EXTRACT_CHAR_CAP:
                out.append(
                    f"[…truncated; {len(wb.sheetnames) - len(out)} "
                    "more sheets skipped]"
                )
                break
            out.append(chunk)
            total_chars += len(chunk) + 2  # "\n\n" separator
    finally:
        wb.close()

    text = "\n\n".join(out)
    return text, len(text)


def extract_txt(path: Path) -> tuple[str, int]:
    """Extract TXT/MD as utf-8-sig (auto-strips BOM, devil L2)."""
    try:
        text = path.read_text(encoding="utf-8-sig", errors="replace")
    except OSError as exc:
        raise ExtractionError(f"read failed: {exc}") from exc
    return text, len(text)


# extract_md is alias-identical to extract_txt.
extract_md = extract_txt


def extract_pdf(path: Path) -> tuple[str, int]:
    """Extract PDF via pypdf - used as Option C fallback ONLY.

    Sub-100-char total -> Russian hint string (devil M8 accepted false
    positive: a real text PDF with <100 chars total triggers the same
    hint).
    """
    try:
        from pypdf import PdfReader
        from pypdf.errors import PdfReadError
    except ImportError as exc:
        raise ExtractionError(f"pypdf not installed: {exc}") from exc

    try:
        reader = PdfReader(str(path))
    except PdfReadError as exc:
        raise ExtractionError(f"corrupt PDF: {exc}") from exc
    except Exception as exc:
        raise ExtractionError(f"PDF parse failed: {exc}") from exc

    if reader.is_encrypted:
        raise ExtractionError("encrypted PDF")

    parts: list[str] = []
    for page in reader.pages:
        try:
            t = page.extract_text() or ""
        except Exception:
            t = ""
        parts.append(t)
    text = "\n\n".join(parts).strip()
    if len(text) < PDF_MIN_TEXT_LAYER_CHARS:
        return (
            "[PDF appears to have no text layer; OCR not available in phase 6a]",
            0,
        )
    return text, len(text)


# Public dispatch table - handler keys on attachment_kind.
EXTRACTORS: dict[str, Callable[[Path], tuple[str, int]]] = {
    "docx": extract_docx,
    "xlsx": extract_xlsx,
    "txt": extract_txt,
    "md": extract_md,
    "pdf": extract_pdf,
}
```

## 6. Boot-sweep (`src/assistant/main.py`)

```python
import time

def _boot_sweep_uploads(uploads_dir: Path) -> None:
    """Phase 6a (devil H3): UNCONDITIONAL boot-sweep.

    Daemon.start() was just (re)started; nothing in uploads_dir/ from
    the previous boot is in-flight. Wipe ALL top-level entries
    (orphaned tmp files); for ``.failed/`` quarantine, prune entries
    older than 7 days. The 1h-bound policy from the prior plan is
    DROPPED - it was open to crash-loop disk fill.
    """
    if not uploads_dir.exists():
        return
    now = time.time()
    for entry in uploads_dir.iterdir():
        if entry.name == ".failed":
            if entry.is_dir():
                for f in entry.iterdir():
                    try:
                        if now - f.stat().st_mtime > 7 * 86400:
                            f.unlink(missing_ok=True)
                    except OSError as exc:
                        log.warning(
                            "boot_sweep_failed_prune_error",
                            path=str(f), error=repr(exc),
                        )
            continue
        try:
            if entry.is_file() or entry.is_symlink():
                entry.unlink(missing_ok=True)
        except OSError as exc:
            log.warning(
                "boot_sweep_orphan_unlink_error",
                path=str(entry), error=repr(exc),
            )
```

**Why no async sweep?** uploads dir is bounded (single owner; 20 MB cap; 7d horizon -> < 1.4 GB worst case). Sync iterdir + unlink is fast (<10 ms typical) and the sweep happens before adapter accepts traffic; no need to background it.

**Periodic prune note:** the original plan suggested an end-of-handler-turn periodic prune. Coder may add a per-turn "sweep my own tmp file if still present" call (already covered by the `finally` in `_handle_locked`). A separate periodic background sweep is **not required for 6a** - the boot-sweep + per-turn cleanup together bound disk usage tightly enough.
